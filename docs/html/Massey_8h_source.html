34  std::vector<std::pair<T, S>> general_canonical(const rank_t& first, const rank_t& second, const std::vector<T>& firstnames, const std::vector<S>& secondnames) {
35  std::vector<std::pair<T, S>> vec;
36  vec.reserve(summation(first) * summation(second));
37  auto s = first.size();
38  auto t = second.size();
39  for (int d = 0; d < t; d++) {
40  auto sumsecond = summation(second, d);
41  for (int c = 0; c < s; c++) {
42  auto sumfirst = summation(first, c);
43  for (int j = 0; j < std::min(first[c], second[d]); j++) {
44  for (int i = 0; i < std::max(first[c], second[d]); i++) {
45  auto pair = std::make_pair(firstnames[(i % first[c]) + sumfirst], secondnames[(i + j) % second[d] + sumsecond]);
46  vec.push_back(pair);
47  }
48  }
49  }
50  }
51  return vec;
52  }

277  void MasseyCompute<rank_t, diff_t>::compute() {
278  auto factor1 = product_bottom(BoxCD, E, Box, boundaryCD, resgenE, degreeC + degreeD + 1, degreeE);
279  gen_t<rank_t, diff_t> factor2 = BoxDE_to_BoxCD * product_bottom(C, BoxDE, C_DE_detailedrank, resgenC, boundaryDE, degreeC, degreeD + degreeE + 1);
280 
281  typename diff_t::Scalar sign;
282  if (degreeC % 2)
283  sign = 1;
284  else
285  sign = -1;
286  gen_t<rank_t, diff_t> Masseyproduct_bottom = factor1 + sign * factor2;
287 
288  ID = IDGeneratorCompute<rank_t, diff_t>(level, Box);
289  Mass.boxID = std::move(ID.ID);
290  Mass.isZero = ID.H_level.isZero;
291  auto Masseyproduct = invRes(Masseyproduct_bottom, Box.rank, ID.rank_level);
292  Mass.basis = ID.H_level.basis(Masseyproduct);
293  Mass.isZero = Mass.basis.isZero();
294  Mass.normalBasis = Mass.basis;
295  normalize(Mass.normalBasis, ID.H_level.Groups);
296  Mass.Groups = ID.H_level.Groups;

250  void MultiplicationTable<rank_t, diff_t>::pass(const std::vector<int>& element_indices, const std::vector<int>& irreducible_indices) {
251  for (const auto& i:element_indices) {
252  edges[i].reserve(2 * irreducible_indices.size());
253  auto maxirr_plus1 = irreducible_indices.back() + 1;
254  auto degree_index = this->tracker[i];
255  int counter = 0;
256  disconnected.clear();
257  for (const auto& j : irreducible_indices)
258  {

260  auto index_product = Greens[j + degree_index * maxirr_plus1].second;

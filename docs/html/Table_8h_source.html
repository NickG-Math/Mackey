56  TableInput<rank_t, diff_t>::TableInput(const std::vector<int>& minsphere, const std::vector<int>& maxsphere) :
57  minsphere(minsphere), maxsphere(maxsphere)
58  {
59  auto spheres = DegreeConstruction(minsphere, maxsphere);
60  auto maxlength = spheres.size() * (dimension(maxsphere - minsphere) + 1);
61  NonZeroHomology.reserve(maxlength);
62  antimap.reserve(maxlength);
63  IndexedChains.resize(spheres.size());
64 
65  for (const auto& i : spheres) {
66  auto C = ROChains<rank_t, diff_t>(i);
67  for (int k = 0; k <= C.maxindex;k++) {
68  Junction<rank_t, diff_t> J(C, k);
69  Junction<rank_t, diff_t> J_top = transfer(J, power);
70  Homology<rank_t, diff_t> H(J_top);
71  if (!H.isZero) {
72  for (int j = 0; j < H.Groups.size(); j++) {
73  NonZeroHomology.push_back(H.Groups); // we want NonZeroHomology to be in sync with antimap and map, this won't happen too many times
74 
75  std::vector<int> generator;
76  generator.reserve(3 + i.size());
77  generator.push_back(invReindex(k, i));
78  generator.insert(generator.end(), i.begin(), i.end());
79  generator.push_back(j);
80  generator.push_back(1);
81  antimap.push_back(generator);
82  map[generator] = antimap.size() - 1;
83  }
84  }
85  }
86  IndexedChains[hash(i)] = std::move(C);
87  }

372 
373  auto degreesum = this->antimap[i] + basicIrreducibles[j];

375  rank_t normalBasis, Groups;

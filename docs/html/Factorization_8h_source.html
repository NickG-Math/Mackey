146  void MultiplicationGraph<rank_t, diff_t>::pass(const std::vector<int>& element_indices, const std::vector<int>& irreducible_indices) {
147  disconnected.clear();
148  for (const auto& i : element_indices) {
149  edges[i].reserve(2 * this->number_of_irreducibles);
150  edgeid[i].reserve(2 * this->number_of_irreducibles);
151  colors[i].reserve(2 * this->number_of_irreducibles);
152 
153  auto maxirr_plus1 = irreducible_indices.back() + 1;
154  auto degree_index = tracker[i];
155  int counter = 0;
156  for (const auto& j : irreducible_indices)
157  {
158  auto G = this->Greens[j + degree_index * maxirr_plus1].first;
159  auto index_product = this->Greens[j + degree_index * maxirr_plus1].second;
160  if (!G.isZero)
161  {
162  auto pair = determine_connection(G, i, j, index_product);
163  auto position = pair.first;
164  auto multiple = pair.second;
165  if (position >= 0 && multiple != 0)
166  {
167  connect(i, j, index_product, position, multiple);
168  counter++;
169  }
170  }
171  }
172  if (counter == 0)
173  disconnected.push_back(i);

184  int multiple = 0;
185  int position = 0;
186  rank_t newbasis;
187  if (!basis.isZero() && basis.size() == 1)
188  multiple = basis[0];

201

206  bool hit = 0;
207  for (const auto& t : elements_i) {
208  hit += (find(edges[t], m) >= 0);
209  }

211  nothit.push_back(k);

213  }

415  template<typename rank_t, typename diff_t>
416  void Factorization<rank_t, diff_t>::factorize(int i) {
417  if (this->path[i].empty())
418  return;
419 
420  factorization.clear();
421  orderOfoperations.clear();
422  multiple = 1;
423 
424  bool flagpos, flagneg;
425  flagpos = flagneg = 0;
426  std::vector<int> poscounter(this->number_of_irreducibles);
427  std::vector<int> negcounter(this->number_of_irreducibles);
428 
429  for (int j = this->path[i].size() - 2; j >= 0; j--) {
430  int indexstart = this->tracker[this->path[i][j]];
431  int indexend= this->tracker[this->path[i][j+1]];
432 
433  auto difference = this->degree[indexstart] - this->degree[indexend];
434  auto difference_last = this->element[this->path[i][j]].back() - this->element[this->path[i][j+1]].back();
435  if (difference_last != 0 && isZero(difference) ) {
436  multiple *= this->element[this->path[i][j]].back();
437  }
438 
439 
440  for (decltype(this->number_of_irreducibles) k = 0; k < this->number_of_irreducibles; k++) {
441  if (difference == this->basicIrreducibles[k]) {
442  if (flagneg) {
443  orderOfoperations.push_back(-1);
444  factorization.push_back(negcounter); //flush
445  negcounter.assign(negcounter.size(), 0);
446  flagneg = 0;
447  }
448  poscounter[k]++;
449  flagpos = 1;
450  break;
451  }
452  else if (difference == - this->basicIrreducibles[k]) {
453  if (flagpos) {
454  orderOfoperations.push_back(1);
455  factorization.push_back(poscounter); //flush
456  poscounter.assign(poscounter.size(), 0);
457  flagpos = 0;
458  }
459  negcounter[k]++;
460  flagneg = 1;
461  break;
462  }
463  }
464  }
465  if (flagpos) {
466  orderOfoperations.push_back(1);
467  factorization.push_back(poscounter);
468  }
469  else if (flagneg) {
470  orderOfoperations.push_back(-1);
471  factorization.push_back(negcounter);
472  }
473  }

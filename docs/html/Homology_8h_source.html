244  void normalize(Eigen::Matrix<T,1,-1>& element, const Eigen::Matrix<T, 1, -1>& group) {
245  if (element.isZero())
246  return;
247  mod_normalize(element,group);
248  if (element.isZero())
249  return;
250  int size = element.size();
251  if (size == 1) {
252  if (group[0] == 1 && element[0] < 0) {
253  element = -element;
254  mod_normalize(element, group);
255  return;
256  }
257  if (group[0]!=1) {
258  element[0] = std::gcd(element[0], group[0]);
259  return;
260  }
261  }
262  auto n = order(element, group);
263  if (n == 0) {
264  for (int i = 0; i < size; i++) {
265  if (group[i] == 1 && element[i] < 0) {
266  element = -element;
267  mod_normalize(element, group);
268  return;
269  }
270  }
271  }
272  if (n <= 2)
273  return;
274  else {
275  auto ideal_element = element;
276  for (int i = 0; i < size; i++) {
277  if (group[i] != 1 && ideal_element[i] != 0)
278  ideal_element[i] = std::gcd(ideal_element[i], group[i]);
279  }
280  int closest_to_ideal = 1;
281  int max_hits = -1;
282  for (int i = 1; i < n; i++) {
283  if (std::gcd(i, n) == 1) {
284  int current_hits = 0;
285  for (int j = 0; j < size; j++) {
286  if ((i * element[j] - ideal_element[j]) % group[j]==0)
287  current_hits++;
288  }
289  if (max_hits == -1 || max_hits < current_hits) {
290  if (current_hits == size) {
291  element = ideal_element;
292  return;
293  }
294  max_hits = current_hits;
295  closest_to_ideal = i;
296  }
297  }
298  }
299  if (closest_to_ideal == 1)
300  return;
301  element=closest_to_ideal * element;
302  mod_normalize(element, group);
303  return;
304  }

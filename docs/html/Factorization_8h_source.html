148  template<typename rank_t, typename diff_t>
149  factorization_triple Factorization<rank_t, diff_t>::factorize(int i) const {
150  factorization_triple f;
151  if (this->path[i].empty())
152  return f;
153  f.factorization.reserve(8); //usually good
154  f.orderOfoperations.reserve(8); //usually good
155  bool flagpos, flagneg;
156  flagpos = flagneg = 0;
157  std::vector<int> poscounter(this->number_of_irreducibles);
158  std::vector<int> negcounter(this->number_of_irreducibles);
159  f.multiple = 1;
160  for (int j = this->path[i].size() - 2; j >= 0; j--) {
161  int indexstart = this->tracker[this->path[i][j]];
162  int indexend = this->tracker[this->path[i][j + 1]];
163 
164  auto difference = this->degree[indexstart] - this->degree[indexend];
165  if (isZero(difference)) {
166  int m = this->isMultiple(this->path[i][j], this->path[i][j + 1]);
167  if (m != 0)
168  f.multiple *= m;
169  }
170  for (int k = 0; k < this->number_of_irreducibles; k++) {
171  if (difference == this->basicIrreducibles[k]) {
172  if (flagneg) {
173  f.orderOfoperations.push_back(-1);
174  f.factorization.push_back(negcounter); //flush
175  negcounter.assign(negcounter.size(), 0);
176  flagneg = 0;
177  }
178  poscounter[k]++;
179  flagpos = 1;
180  break;
181  }
182  else if (difference == -this->basicIrreducibles[k]) {
183  if (flagpos) {
184  f.orderOfoperations.push_back(1);
185  f.factorization.push_back(poscounter); //flush
186  poscounter.assign(poscounter.size(), 0);
187  flagpos = 0;
188  }
189  negcounter[k]++;
190  flagneg = 1;
191  break;
192  }
193  }
194  }
195  if (flagpos) {
196  f.orderOfoperations.push_back(1);
197  f.factorization.push_back(poscounter);
198  }
199  else if (flagneg) {
200  f.orderOfoperations.push_back(-1);
201  f.factorization.push_back(negcounter);
202  }
203  return f;
204  }

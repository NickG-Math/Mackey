165  void Factorization<rank_t, diff_t>::factorize(int i) {
166  if (this->path[i].empty())
167  return;
168 
169  factorization.clear();
170  orderOfoperations.clear();
171 
172  bool flagpos, flagneg;
173  flagpos = flagneg = 0;
174  std::vector<int> poscounter(this->number_of_irreducibles);
175  std::vector<int> negcounter(this->number_of_irreducibles);
176 
177  multiple = 1;
178  for (int j = this->path[i].size() - 2; j >= 0; j--) {
179  int indexstart = this->tracker[this->path[i][j]];
180  int indexend = this->tracker[this->path[i][j + 1]];
181 
182  auto difference = this->degree[indexstart] - this->degree[indexend];
183  if (isZero(difference)) {
184  int m = this->isMultiple(this->path[i][j], this->path[i][j + 1]);
185  if (m != 0)
186  multiple *= m;
187  }
188  for (int k = 0; k < this->number_of_irreducibles; k++) {
189  if (difference == this->basicIrreducibles[k]) {
190  if (flagneg) {
191  orderOfoperations.push_back(-1);
192  factorization.push_back(negcounter); //flush
193  negcounter.assign(negcounter.size(), 0);
194  flagneg = 0;
195  }
196  poscounter[k]++;
197  flagpos = 1;
198  break;
199  }
200  else if (difference == -this->basicIrreducibles[k]) {
201  if (flagpos) {
202  orderOfoperations.push_back(1);
203  factorization.push_back(poscounter); //flush
204  poscounter.assign(poscounter.size(), 0);
205  flagpos = 0;
206  }
207  negcounter[k]++;
208  flagneg = 1;
209  break;
210  }
211  }
212  }
213  if (flagpos) {
214  orderOfoperations.push_back(1);
215  factorization.push_back(poscounter);
216  }
217  else if (flagneg) {
218  orderOfoperations.push_back(-1);
219  factorization.push_back(negcounter);
220  }
221  }

121  void Factorization<rank_t, diff_t>::factorize(int i) {
122  if (this->path[i].empty())
123  return;
124 
125  factorization.clear();
126  orderOfoperations.clear();
127 
128  bool flagpos, flagneg;
129  flagpos = flagneg = 0;
130  std::vector<int> poscounter(this->number_of_irreducibles);
131  std::vector<int> negcounter(this->number_of_irreducibles);
132 
133  multiple = 1;
134  for (int j = this->path[i].size() - 2; j >= 0; j--) {
135  int indexstart = this->tracker[this->path[i][j]];
136  int indexend = this->tracker[this->path[i][j + 1]];
137 
138  auto difference = this->degree[indexstart] - this->degree[indexend];
139  if (isZero(difference)) {
140  int m = this->isMultiple(this->path[i][j], this->path[i][j + 1]);
141  if (m != 0)
142  multiple *= m;
143  }
144  for (int k = 0; k < this->number_of_irreducibles; k++) {
145  if (difference == this->basicIrreducibles[k]) {
146  if (flagneg) {
147  orderOfoperations.push_back(-1);
148  factorization.push_back(negcounter); //flush
149  negcounter.assign(negcounter.size(), 0);
150  flagneg = 0;
151  }
152  poscounter[k]++;
153  flagpos = 1;
154  break;
155  }
156  else if (difference == -this->basicIrreducibles[k]) {
157  if (flagpos) {
158  orderOfoperations.push_back(1);
159  factorization.push_back(poscounter); //flush
160  poscounter.assign(poscounter.size(), 0);
161  flagpos = 0;
162  }
163  negcounter[k]++;
164  flagneg = 1;
165  break;
166  }
167  }
168  }
169  if (flagpos) {
170  orderOfoperations.push_back(1);
171  factorization.push_back(poscounter);
172  }
173  else if (flagneg) {
174  orderOfoperations.push_back(-1);
175  factorization.push_back(negcounter);
176  }
177  }

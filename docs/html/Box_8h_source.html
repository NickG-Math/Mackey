114  Derived permutation_block(const Eigen::PermutationMatrix<-1, -1, S>& left, const Eigen::PermutationMatrix<-1, -1, S>& right, const Eigen::MatrixBase<Derived>& a, int k) {
115  int n = a.cols();
116  int m = a.rows();
117  Derived b = Eigen::MatrixBase<Derived>::Zero(m * k, n * k);

137  auto rows = L[0].rows();

169  void BoxPoint<rank_t, diff_t> ::getdiff() {
170  std::vector<diff_t> LeftDiff(i + 1);
171  std::vector<diff_t> RightDiff(i + 1);
172 
173  for (int j = lowlimit; j <= highlimit; j++) {
174  auto Domain = memoChangeBasis<rank_t>(C.rank[j], D.rank[i - j]);
175  if (j >= 1) //We have a LeftDiff
176  {
177  auto RangeL = memoChangeBasis<rank_t>(C.rank[j - 1], D.rank[i - j]);
178  LeftDiff[j] = permutation_block(RangeL.LefttoCanon, Domain.LefttoCanon, C.diff[j], summation(D.rank[i - j]));
179  }
180  if (i - j >= 1) //We have a RightDiff
181  {
182  typename diff_t::Scalar sign = (1 - 2 * (j % 2)); //(1 - 2 * (j % 2)) is(-1) ^ j
183  auto RangeR = memoChangeBasis<rank_t>(C.rank[j], D.rank[i - j - 1]);
184  RightDiff[j] = permutation_block(RangeR.RighttoCanon, Domain.RighttoCanon, static_cast<diff_t>(sign * D.diff[i - j]), summation(C.rank[j]));
185  }
186  }
187  diff = MatrixMixer(LeftDiff, RightDiff);

178  void WeightedGraph::stepDistance(int start) {

180  auto i = edges[start][j];
181  if (!visited[i]) {
182  if (distance[i] == -1 || distance[i] > distance[start] + weights[start][j]) {
183  distance[i] = distance[start] + weights[start][j];

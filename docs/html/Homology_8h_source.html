244  if (element.isZero())
245  return;

252  element = -element;
253  mod_normalize(element, group);

255  }

258  return;
259  }
260  }

262  if (n == 0) {
263  for (int i = 0; i < size; i++) {
264  if (group[i] == 1 && element[i] < 0) {
265  element = -element;
266  mod_normalize(element, group);
267  return;
268  }
269  }
270  }
271  if (n <= 2)
272  return;
273  else {
274  auto ideal_element = element;
275  for (int i = 0; i < size; i++) {
276  if (group[i] != 1 && ideal_element[i] != 0)
277  ideal_element[i] = std::gcd(ideal_element[i], group[i]);
278  }
279  int closest_to_ideal = 1;
280  int max_hits = -1;
281  for (int i = 1; i < n; i++) {
282  if (std::gcd(i, n) == 1) {
283  int current_hits = 0;
284  for (int j = 0; j < size; j++) {
285  if ((i * element[j] - ideal_element[j]) % group[j]==0)
286  current_hits++;
287  }
288  if (max_hits == -1 || max_hits < current_hits) {
289  if (current_hits == size) {

291  return;
292  }
293  max_hits = current_hits;
294  closest_to_ideal = i;
295  }
296  }
297  }

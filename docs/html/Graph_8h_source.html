224  void WeightedGraph::stepDistance(int start) {
225  for (size_t j = 0; j < edges[start].size(); j++) {
226  auto i = edges[start][j];
227  if (!visited[i]) {
228  if (distance[i] == -1 || distance[i] > distance[start] + weights[start][j] + 1) {
229  distance[i] = distance[start] + weights[start][j] + 1;
230  unweighted_distance[i] = unweighted_distance[start] + 1;
231  closest[i] = start;
232  next.push(std::make_pair(-distance[i], i));
233  }
234  }
235  }
236  visited[start] = 1;
237  reachable[start] = 1;

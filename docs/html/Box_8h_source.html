123  LeftDiff.resize(i + 1);

126  for (int j = lowlimit; j <= highlimit; j++) {
127  auto Domain = memoChangeBasis<rank_t>(C.rank[j], D.rank[i - j]);
128  if (j >= 1) //We have a LeftDiff
129  {
130  diff_t convLeftDiff = blkdiag(C.diff[j], summation(D.rank[i - j]));
131  auto RangeL = memoChangeBasis<rank_t>(C.rank[j - 1], D.rank[i - j]);
132  LeftDiff[j] = RangeL.LefttoCanon.inverse() * convLeftDiff * Domain.LefttoCanon;
133  }
134  if (i - j >= 1) //We have a RightDiff

136  auto sign = (1 - 2 * (j % 2)); //(1 - 2 * (j % 2)) is(-1) ^ j

139  RightDiff[j] = RangeR.RighttoCanon.inverse() * convRightDiff * Domain.RighttoCanon;
140  }

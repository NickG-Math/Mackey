44  rank_t rankmult(const rank_t& A, const rank_t& B) {
45  int sum = 0;
46  for (int i = 0; i < A.size(); i++) {
47  for (int j = 0; j < B.size(); j++) {
48  sum += std::min(A(i), B(j));
49  }
50  }
51  rank_t C(sum);
52  int current = 0;
53  for (int j = 0; j < B.size(); j++) {
54  for (int i = 0; i < A.size(); i++) {
55  C.segment(current, std::min(A(i), B(j))).setConstant(std::max(A(i), B(j)));
56  current += std::min(A(i), B(j));
57  }
58  }
59  return C;

116  Derived permutation_block(const Eigen::PermutationMatrix<-1, -1, S>& left, const Eigen::PermutationMatrix<-1, -1, S>& right, const Eigen::MatrixBase<Derived>& a, int k) {
117  int n = a.cols();
118  int m = a.rows();
119  Derived b = Eigen::MatrixBase<Derived>::Zero(m * k, n * k);
120  for (int i = 0; i < b.rows(); i++) {
121  auto u = left.indices()[i] % m;

123  b.row(i).segment(n * v, n) = a.row(u).segment(0, n);

176  template<typename rank_t, typename diff_t>
177  void BoxPoint<rank_t, diff_t> ::getdiff() {
178 
179  if constexpr (SFINAE::is_Dense<diff_t>::value) {//dense implementation
180 
181  std::vector<diff_t> LeftDiff(i + 1);
182  std::vector<diff_t> RightDiff(i + 1);
183  for (int j = lowlimit; j <= highlimit; j++) {
184  auto Domain = ChangeBasis<int>(C.rank[j], D.rank[i - j]);
185  if (j >= 1) //We have a LeftDiff
186  {
187  auto RangeL = ChangeBasis<int>(C.rank[j - 1], D.rank[i - j]);
188  LeftDiff[j] = permutation_block(RangeL.LefttoCanon, Domain.LefttoCanon, C.diff[j], summation(D.rank[i - j]));
189  }
190  if (i - j >= 1) //We have a RightDiff
191  {
192  typename diff_t::Scalar sign = (1 - 2 * (j % 2)); //(1 - 2 * (j % 2)) is(-1) ^ j
193  auto RangeR = ChangeBasis<int>(C.rank[j], D.rank[i - j - 1]);
194  RightDiff[j] = permutation_block(RangeR.RighttoCanon, Domain.RighttoCanon, static_cast<diff_t>(sign * D.diff[i - j]), summation(C.rank[j]));
195  }
196  }
197  diff = MatrixMixer(LeftDiff, RightDiff, summation(rank_range), summation(rank_domain));
198 
199  }
200  else {//sparse implementation
201  typename diff_t::StorageIndex totalsize, v_offset, h_offset;
202  totalsize = v_offset = h_offset = 0;
203 
204  //first find the size of nonzeros
205  for (int j = lowlimit; j <= highlimit; j++) {
206  if (j >= 1)
207  totalsize += C.diff[j].nonZeros()* summation(D.rank[i - j]);
208  if (i - j >= 1)
209  totalsize += D.diff[i - j].nonZeros() * summation(C.rank[j]);
210  }
211 
212  triplets<Scalar_t<diff_t>, typename diff_t::StorageIndex> mat;
213  mat.reserve(totalsize);
214 
215  //put the nonzeros in mat
216  for (int j = lowlimit; j <= highlimit; j++) {
217  auto Domain = ChangeBasis<typename diff_t::StorageIndex>(C.rank[j], D.rank[i - j], 0);
218  if (j >= 1) //We have a LeftDiff
219  {
220  auto RangeL = ChangeBasis<typename diff_t::StorageIndex>(C.rank[j - 1], D.rank[i - j], 0);
221  box_sparse(mat, RangeL.LefttoCanon, Domain.LefttoCanon, make_triplets(C.diff[j]), C.diff[j].rows(), C.diff[j].cols(), summation(D.rank[i - j]), h_offset, v_offset);
222  v_offset += C.diff[j].rows() * summation(D.rank[i - j]);
223  }
224  if (i - j >= 1) //We have a RightDiff
225  {
226  typename diff_t::Scalar sign = (1 - 2 * (j % 2)); //(1 - 2 * (j % 2)) is(-1) ^ j
227  auto RangeR = ChangeBasis<typename diff_t::StorageIndex>(C.rank[j], D.rank[i - j - 1], 0);
228  box_sparse(mat, RangeR.RighttoCanon, Domain.RighttoCanon, make_triplets(static_cast<diff_t>(sign * D.diff[i - j])), D.diff[i - j].rows(), D.diff[i - j].cols(), summation(C.rank[j]), h_offset, v_offset);
229  h_offset += D.diff[i - j].cols() * summation(C.rank[j]);
230  }
231  }
232  //turn mat to a matrix
233  diff.resize(summation(rank_range), summation(rank_domain));
234  diff.setFromTriplets(mat.begin(), mat.end());
235  }

144  bool flagpos, flagneg;

154  auto difference = this->degree[indexstart] - this->degree[indexend];
155  if (isZero(difference)) {

157  if (m != 0)

159  }

163  orderOfoperations.push_back(-1);
164  factorization.push_back(negcounter); //flush
165  negcounter.assign(negcounter.size(), 0);
166  flagneg = 0;
167  }
168  poscounter[k]++;
169  flagpos = 1;
170  break;
171  }
172  else if (difference == -this->basicIrreducibles[k]) {
173  if (flagpos) {
174  orderOfoperations.push_back(1);
175  factorization.push_back(poscounter); //flush
176  poscounter.assign(poscounter.size(), 0);
177  flagpos = 0;
178  }
179  negcounter[k]++;
180  flagneg = 1;
181  break;
182  }
183  }
184  }
185  if (flagpos) {
186  orderOfoperations.push_back(1);
187  factorization.push_back(poscounter);
188  }
189  else if (flagneg) {
